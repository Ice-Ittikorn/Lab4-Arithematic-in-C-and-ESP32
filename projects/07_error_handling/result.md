bool validate_thai_id(const char* id) {
    if (strlen(id) != 13) return false;
    int sum = 0;
    for (int i = 0; i < 12; ++i) {
        if (id[i] < '0' || id[i] > '9') return false;
        sum += (id[i] - '0') * (13 - i);

}
